def call_ollama(model, system_prompt, user_prompt, timeout=120):
    """Call Ollama chat API. Returns response text."""
    payload = {
        "model": model,
        "messages": [
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": user_prompt}
        ],
        "stream": False,
        "options": {"temperature": 0.1}
    }

        f"{OLLAMA_URL}/api/chat",
        json=payload,
        timeout=timeout
    )

def extract_json(text):
    """Try to parse JSON from text. Handle markdown wrapping."""
    text = text.strip()
    # Try direct parse
    try:
        return json.loads(text)
    except json.JSONDecodeError:
        pass
    # Try extracting from markdown code block
    import re
    match = re.search(r'```(?:json)?\s*(\{.*?\})\s*```', text, re.DOTALL)
    if match:
        try:
            return json.loads(match.group(1))
        except json.JSONDecodeError:
            pass
    # Try finding any JSON object
    match = re.search(r'\{.*\}', text, re.DOTALL)
    if match:
        try:
            return json.loads(match.group(0))
        except json.JSONDecodeError:
            pass
    raise ValueError(f"Could not extract JSON from: {text[:500]}")

def run_cycle():
    """Execute one complete training cycle."""
    state = load_state()
    
    log(f"\n{'='*60}")
    log(f"CYCLE {state['current_cycle']} STARTING")
    log(f"Curriculum stage: {state['curriculum_stage']}/12")
    log(f"{'='*60}")
    
    # Stage 1: Task assignment
    task = stage1_task_assignment(state)
    
    # Stage 2: Student attempt
    student_code = stage2_student_attempt(task)
    
    # Stage 3: Grading
    grading, execution_output = stage3_teacher_grading(task, student_code, state)
    
    if not grading.get("passed", False):
        log("CYCLE FAILED - Student must retry same task next cycle")
        state["total_cycles_completed"] += 1
        state["current_cycle"] += 1
        save_state(state)
        generate_progress_report(state)
        return False
    
    # Stage 4: Sandbox break
    stage4_sandbox_break(grading, state)
    
    # Stage 5: Improvement demonstration
    growth = stage5_improvement(task, grading, state)
    
    state["total_cycles_completed"] += 1
    state["current_cycle"] += 1
    save_state(state)
    
    generate_progress_report(state)
    
    log(f"CYCLE {state['current_cycle']-1} COMPLETE - PASSED")
    return True
